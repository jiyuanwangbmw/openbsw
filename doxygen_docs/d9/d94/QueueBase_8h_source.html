67  uint32_t size() const
68  {
69  // Volatile is needed to to make sure the read is not optimized away when new elements have
70  // been added to the queue by the other core.
71  uint32_t const txPos = static_cast<uint32_t const volatile&>(_sent);
72  return (txPos >= _received) ? (txPos - _received) : (txPos + (2U * _maxSize)) - _received;
73  }
74 
76  bool isFull() const
77  {
78  // Volatile is needed to make sure the read is not optimized away when isFull() is called
79  // in a loop like `while(sender.isFull()){}`.
80  return (
81  _sent
82  == ((static_cast<uint32_t const volatile&>(_received) + _maxSize) % (2U * _maxSize)));
83  }
84 
86  bool isEmpty() const
87  {
88  // Volatile is needed to prevent `isEmpty()` from returning `true` when in fact new elements
89  // have been added to the queue by the other core.
90  return static_cast<uint32_t const volatile&>(_sent) == _received;
91  }

100  void takeSnapshot()
101  {
102  uint32_t const currentSize = size();
103  if (0U != currentSize)
104  {
105  _stats.loadSnapshot += static_cast<uint16_t>(currentSize);
106  ++_stats.processingCounter;
107  }
108  _stats.realLoadSnapshot += static_cast<uint16_t>(currentSize);
109  ++_stats.realProcessingCounter;
110  if (_stats.previousSnapshot == 0U)
111  {
112  _stats.maxFillRate = static_cast<uint8_t>(currentSize);
113  _stats.previousSnapshot = static_cast<uint8_t>(currentSize);
114  }
115  else
116  {
117  if (currentSize > _stats.previousSnapshot)
118  {
119  auto const diff = (currentSize - _stats.previousSnapshot);
120  if (diff > _stats.maxFillRate)
121  {
122  _stats.maxFillRate = static_cast<uint8_t>(diff);
123  }
124  }
125  _stats.previousSnapshot = static_cast<uint8_t>(currentSize);
126  }
127  }

152  etl::optional<size_t> writeNext()
153  {
154  etl::optional<size_t> writableIndex{};
155  if (isFull())
156  {
157  ++_stats.lostMessages;
158  }
159  else
160  {
161  writableIndex.emplace(_sent % _maxSize);
162  _sent = (_sent + 1U) % (2U * _maxSize);
163  if (size() > _stats.maxLoad)
164  {
165  _stats.maxLoad = static_cast<uint8_t>(size());
166  }
167  if (0U == _stats.processedMessages)
168  {
169  ++_stats.startupLoad;
170  }
171  }
172 
173  return writableIndex;
174  }
